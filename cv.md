    <div class="counter">
        <h1>Counter: <span id="count">0</span></h1>
        <button onclick="increment()">Increment</button>
        <button onclick="decrement()">Decrement</button>
    </div>

    <script>
        let count = 0;
        const countDisplay = document.getElementById('count');

        function increment() {
            count++;
            countDisplay.textContent = count;
        }

        function decrement() {
            count--;
            countDisplay.textContent = count;
        }
    </script>
</body>
</html>
